<script>
  // Update prayer times
  function updatePrayerTimes() {
    document.getElementById('fajr').innerText = 'Fajr: ' + document.getElementById('editFajr').value;
    document.getElementById('zuhr').innerText = 'Zuhr: ' + document.getElementById('editZuhr').value;
    document.getElementById('asr').innerText = 'Asr: ' + document.getElementById('editAsr').value;
    document.getElementById('maghrib').innerText = 'Maghrib: ' + document.getElementById('editMaghrib').value;
    document.getElementById('isha').innerText = 'Isha: ' + document.getElementById('editIsha').value;
    alert('Prayer times updated!');
  }

  // Update About Masjid text
  function updateAbout() {
    document.getElementById('aboutText').innerText = document.getElementById('editAbout').value;
    alert('About Masjid updated!');
  }

  // Update Donate message
  function updateDonate() {
    document.getElementById('donateText').innerText = document.getElementById('editDonate').value;
    alert('Donate message updated!');
  }

  // Add new image to gallery
  function addImage() {
    const url = document.getElementById('newImageURL').value;
    if(url) {
      const img = document.createElement('img');
      img.src = url;
      document.getElementById('galleryImages').appendChild(img);
      document.getElementById('newImageURL').value = '';
      alert('Image added!');
    } else {
      alert('Please enter an image URL.');
    }
  }

  // Language selection (placeholder, you can expand for full translations)
  function changeLanguage() {
    const lang = document.getElementById('languageSelect').value;
    if(lang==='hi'){ alert('Hindi language selected (you can add translations)'); }
    else if(lang==='ur'){ alert('Urdu language selected (you can add translations)'); }
    else{ alert('English selected'); }
  }
</script>
